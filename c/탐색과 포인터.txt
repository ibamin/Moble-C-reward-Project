이진탐색 : 정렬된 배열에서 중앙값을 찾아 비교하여 찾음
예시 코드
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <math.h>
#define size 16

int binary_search(int arr[], int bottom, int top, int target) {
	int m = (bottom + top) / 2;
	if (arr[m] == target) return m;
	if (arr[m] > target) return binary_search(arr, bottom, m-1, target);
	else if(arr[m] < target) return binary_search(arr, m+1, top, target);
	return -1;
}

int main() {
	int key;
	int grade[size] = { 2,6,11,13,18,20,22,27,29,30,34,38,41,42,45,47 };
	printf("찾을 값 :");
	scanf("%d", &key);
	printf("탐색 결과= %d\n", binary_search(grade, 0,size-1, key));
}

2차원 배열
int arr [행][열];
int arr [x][y][z]; // 3차원
예시 코드
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <math.h>
#define row 3
#define cols 5

int main() {
	int arr[row][cols];
	srand(time(NULL));

	for (int i = 0; i < row; i++)
		for (int j = 0; j < cols; j++)
			arr[i][j] = rand() % 100;

	for (int i = 0; i < row; i++) {
		for (int j = 0; j < cols; j++)
			printf("%02d ", arr[i][j]);
		printf("\n");
	}
}
행의 개수는 지정하지 않을 수 있지만 열의 개수는 반드시 지정한다

call by value : 기존에 쓰던 사용자 정의함수의 매개변수
call by reference : 함수에 넘긴 변수의 값을 변경하는 방법
예시 코드
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <math.h>
#define row 3
#define col 6

void swap(int* a, int* b) {
	int tmp = *a;
	*a = *b;
	*b = tmp;
}

int main() {
	int a = 100, b = 200;
	printf("a=%d b=%d\n", a, b);

	swap(&a, &b);

	printf("a=%d b=%d\n", a, b);
}

int get_line_parameter(int a[], float* slope, float* yintercept) {
	if (a[0] == a[2]) return -1;
	else {
		*slope = (float)(a[3] - a[1]) / (float)(a[2] - a[0]);
		*yintercept = a[1] - (*slope) * a[0];
		return 0;
	}
}

int main() {
	float s, y;
	int a[4];
	printf("x1 y1 x2 y2 순으로 입력:");
	for (int i = 0; i < 4; i++) {
		scanf("%d", &a[i]);
	}
	if (get_line_parameter(a[0], a[1], a[2], a[3], &s, &y) == -1) printf("에러\n");
	else printf("기울기는 %f y절편은 %f\n", s, y);
}

void array_copy(int a[], int b[], int size) {
	for (int i = 0; i < size; i++) {
		b[i] = a[i];
	}
}

void array_print(int a[],int size) {
	for (int i = 0; i < size; i++) printf("%d ", a[i]);
	printf("\n");
}

int main() {
	int a[] = { 1,2,3,4,5 };
	int b[5];

	array_copy(a, b, 5);
	array_print(a,5);
	array_print(b, 5);
}
